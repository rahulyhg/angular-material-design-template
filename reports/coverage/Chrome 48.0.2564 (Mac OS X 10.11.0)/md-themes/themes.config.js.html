        var activeTheme;
        activeTheme = 'Walmart';
        mdThemeProvider.globalProvider = $mdThemingProvider;
        mdThemeProvider.activeTheme = activeTheme;
        $mdThemingProvider.setDefaultTheme(activeTheme);
