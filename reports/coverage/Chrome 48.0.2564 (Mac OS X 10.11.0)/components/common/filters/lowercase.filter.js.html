'use strict';
 
/*global angular */
(function () {
    'use strict';
 
    angular.module('lowercase.filter', [])
    /**
     *replace uppercase to regular case
     */
    .filter('humanizeDoc', function () {
        return function (doc) {
            if (!doc) {
                return;
            }
            if (doc.type === 'directive') {
                return doc.name.replace(/([A-Z])/g, function ($1) {
                    return '-' + $1.toLowerCase();
                });
            }
            return doc.label || doc.name;
        };
    });
})();
