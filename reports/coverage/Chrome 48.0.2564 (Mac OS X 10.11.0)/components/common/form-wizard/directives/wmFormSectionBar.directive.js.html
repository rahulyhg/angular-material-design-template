'use strict';
 
/*global angular */
(function () {
    'use strict';
 
    angular.module('formWizard').directive('wmFormSectionBar', wmFormSectionBar);
    wmFormSectionBar.$inject = ['mdThemeProvider', 'walmartBackground', 'asdaBackground'];
    function wmFormSectionBar(mdThemeProvider, walmartBackground, asdaBackground) {
        return {
            restrict: 'AE',
            link: linkFunc
        };
 
        function linkFunc(scope, element) {
            var color;
            if (mdThemeProvider.activeTheme === 'Walmart') {
                color = walmartBackground['A700'];
            } else if (mdThemeProvider.activeTheme === 'ASDA') {
                color = asdaBackground['A700'];
            }
            element.css('background-color', color);
            element.css('color', '#000000');
        }
    }
})();
