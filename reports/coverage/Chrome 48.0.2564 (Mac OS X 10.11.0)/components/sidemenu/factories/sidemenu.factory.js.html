            pages: [{
                name: 'Cheetos',

                icon: 'fa fa-life-ring',
                description: 'Donuts Page'
            }]
        });
 
        function _isSectionSelected(section) {
            if (section) {
                return section.isOpen;
            }
        }
 
        function toggleSection(section) {
            if (section) {
                if (self.openedSection === section) {
                    self.openedSection.isOpen = !self.openedSection.isOpen;
                } else if (!self.openedSection) {
                    self.openedSection = section;
                    self.openedSection.isOpen = !self.openedSection.isOpen;
                } else {
                    section.isOpen = !section.isOpen;
                }
            }
        }
 
        function _selectPage(section, page) {
            if (page && page.url && $location.path(page.url)) {
                self.currentSection = section;
                self.currentPage = page;
            }
        }
 
        self = {
            sections: sections,
 
            toggleSelectSection: function toggleSelectSection(section) {
                toggleSection(section);
            },
            isSectionSelected: function isSectionSelected(section) {
                return _isSectionSelected(section);
            },
 
            selectPage: function selectPage(section, page) {
                if (section && page) {
                    _selectPage(section, page);
                }
            }
        };
 
        return self;
    }]);
})();
